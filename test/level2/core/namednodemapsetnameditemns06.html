     <metadata>
          <title>namednodemapsetnameditemns06</title>
          <creator>IBM</creator>
          <description> Retreieve the first element whose localName is address and its attributes into a named node map. Retreiving the domestic attribute from the namednodemap. Retreieve the second element whose localName is address and its attributes into a named node map. Invoke setNamedItemNS on the second NamedNodeMap specifying the first domestic attribute from the first map. This should raise an INUSE_ATTRIBIUTE_ERR. </description>
          <contributor>Neil Delima</contributor>
          <date qualifier='created'>2002-04-28</date>
          <subject resource='http://www.w3.org/TR/DOM-Level-2-Core/core#ID-setNamedItemNS'/>
</metadata>

     <var name='attributes' type='NamedNodeMap'/>
     <var name='elementList' type='NodeList'/>
     <var name='element' type='Element'/>
     <var name='attr' type='Attr'/>
     <var name='newNode' type='Node'/>
     <load var='doc' href='staffNS' willBeModified='true'/>
     <getElementsByTagNameNS var='elementList' obj='doc' namespaceURI='"*"' localName='"address"' interface='Document'/>
