     <var name='domImpl' type='DOMImplementation'/>
     <var name='hasCore' type='boolean'/>
     <var name='hasXML' type='boolean'/>
     <var name='hasEvents' type='boolean'/>
     <var name='hasLS' type='boolean'/>

                    <isTrue value='hasXML'/>
