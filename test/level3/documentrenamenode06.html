     <var name='attr' type='Attr'/>
     <var name='childList' type='NodeList'/>
     <var name='renamedclass' type='Node'/>
     <var name='nodeName' type='DOMString'/>
     <var name='nodeType' type='int'/>
     <var name='namespaceURI' type='DOMString'/>
