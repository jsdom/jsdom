     <metadata>
          <title>elementReplaceExistingAttribute</title>
          <creator>NIST</creator>
          <description> The "setAttributeNode(newAttr)" method adds a new attribute to the Element. If the "newAttr" Attr node is already present in this element, it should replace the existing one. Retrieve the last child of the third employee and add a new attribute node by invoking the "setAttributeNode(new Attr)" method. The new attribute node to be added is "street", which is already present in this element. The method should replace the existing Attr node with the new one. This test uses the "createAttribute(name)" method from the Document interface. </description>
          <contributor>Mary Brady</contributor>
          <date qualifier='created'>2001-08-17</date>
</metadata>
     <var name='doc' type='Document'/>
     <var name='elementList' type='NodeList'/>
     <var name='testEmployee' type='Element'/>
     <var name='newAttribute' type='Attr'/>
     <var name='name' type='DOMString'/>
     <var name='setAttr' type='Attr'/>
     <load var='doc' href='staff' willBeModified='true'/>
     <getElementsByTagName interface='Document' obj='doc' tagname='"address"' var='elementList'/>
     <item interface='NodeList' obj='elementList' index='2' var='testEmployee'/>
     <createAttribute obj='doc' var='newAttribute' name='"street"'/>
     <setAttributeNode var='setAttr' obj='testEmployee' newAttr='newAttribute'/>
     <getAttribute obj='testEmployee' var='name' name='"street"'/>
     <assertEquals actual='name' expected='""' id='elementReplaceExistingAttributeAssert' ignoreCase='false'/>
</test>
